  data = Clean_data, 
  mapping = aes(
    x = CPIA_Mean
  )
) +
  geom_jitter(mapping = aes(colour = Country, y = model_1)) +
  geom_jitter(mapping = aes(y = GDP_growth)) +
  theme_minimal() +
  labs(title = "Model 1 for predicting GDP growth - Adjusted R-squared:  0.006") +
  xlab("CPIA Mean") +
  ylab("GDP Growth") +
  ylim(-10, 10) 

ggsave(filename = "graphs/model_1.png",
